Les tables des matières


<!-- HTML -->
<ul>
  <li>
    <a href="#title1"></a>
  </li>
  <li>
    <a href="#title2"></a>
  </li>
</ul>

<h2 id="title1">Titre 1</h2>
<h2 id="title2">Titre 2</h2>
    

/* CSS */
a::before {
  content: target-text(attr(href));
}

a::after {
  content: leader(dotted)
   target-counter(attr(href), page);
}

Les marges de pages


<!-- HTML -->

<p>Encore du texte</p>
    

/* CSS */

@page {
  margin: 4cm 2cm 3cm 2cm;
  @top-left {
    content: element(image-header);
  }
}

img {
  position: running(image-header);
}
